001package edu.cnm.deepdive.security;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.InputStream;
006import java.io.InputStreamReader;
007import java.util.ArrayList;
008import java.util.Properties;
009import java.util.Random;
010
011/**
012 * Wordlist set for passpharse and passwords.
013 * @author Yolanda Philgreen
014 *
015 */
016public class Wordlist {
017
018  private static final int RECOMMENDED_PHRASE_LENGTH = 5;
019
020  public static final String WORD_LIST_FILE = "resources/eff_large_wordlist .txt";
021  private static final String PROPERTIES_FILE = "resources/text.properties";
022
023  private static String usageMessage;
024  private static String errorMessage;
025  private static String warningMessage;
026
027  /**
028   * command line to open wordlist using random and phrase specific details.
029   * @param args 
030   * 
031   */
032  public static void main(String[] args) {
033    try {
034      loadResources();
035      int phraseLength = (args.length > 0) ? Integer.parseInt(args[0]) : RECOMMENDED_PHRASE_LENGTH;
036      if (phraseLength <= 0) {
037        throw new IllegalArgumentException(errorMessage);
038      } else if (phraseLength < RECOMMENDED_PHRASE_LENGTH) {
039        System.out.println(warningMessage);
040      }
041      String[] wordList = loadWordList(WORD_LIST_FILE);
042      String[] selectedWords = getRandomWords(phraseLength, wordList);
043      System.out.println(getJoinedString(selectedWords));
044    } catch (NumberFormatException ex) {
045      ex.printStackTrace();
046      System.out.println(errorMessage);
047      System.out.println(usageMessage);
048      System.exit(1);
049    } catch (IllegalArgumentException ex) {
050      System.out.println(errorMessage);
051    } catch (IOException ex) {
052      ex.printStackTrace();
053      System.exit(1);
054
055    }
056
057  }
058
059  private static void loadResources() throws IOException {
060    Properties properties = new Properties();
061    try (InputStream input = Wordlist.class.getClassLoader().getResourceAsStream(PROPERTIES_FILE)) {
062      properties.load(input);
063      usageMessage = properties.getProperty("usage.message");
064      errorMessage = properties.getProperty("error.message");
065      warningMessage = properties.getProperty("warning.message");
066
067    }
068  }
069
070  private static String[] loadWordList(String listPath)
071      throws IOException {
072    try (BufferedReader reader = new BufferedReader(
073        new InputStreamReader(Wordlist.class.getClassLoader().getResourceAsStream(listPath)))) {
074      ArrayList<String> words = new ArrayList<>();
075      for (String line = reader.readLine(); line != null; line = reader.readLine()) {
076        words.add(line.split("\\s+")[1]);
077      }
078      return words.toArray(new String[] {});
079    }
080  }
081
082  /**
083   * 
084   * @param numWords Selection of wordlist amount within random field
085   * @param wordlist 
086   * @return the streamlined password from wordlist
087   */
088  public static String[] getRandomWords(int numWords, String[] wordlist) {
089    String[] selection = new String[numWords];
090    Random rng = new Random();
091    for (int i = 0; i < selection.length; i++) {
092      int selectedPosition = rng.nextInt(wordlist.length);
093      selection[i] = wordlist[selectedPosition];
094    }
095    return selection;
096
097  }
098
099  private static String getJoinedString(String[] source) {
100    StringBuilder builder = new StringBuilder();
101    for (String item : source) {
102      builder.append(item);
103      builder.append(" ");
104    }
105    return builder.toString().trim();
106  }
107
108
109}
