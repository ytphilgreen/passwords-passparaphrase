012/**
013 * A program that generates random passwords and passphrases.
014 * 
015 * Generation uses a cryptographically secure random number generator
016 * to select words from a list or characters from an pool.
017 * 
018 * @author Yolanda Philgreen
019 *
020 */
021public class Guard {
022
023  /**
024   * Parse command line arguments to a parser from the Apache Commons CLI library,
025   * then instantiate and invoke the appropriate classes and methods to 
026   * generate the requested artifact. 
027   * 
028   * @param args   Command line arguments, specifying generation options. 
029   */
030  public static void main(String[] args) {
031  HashMap<String, Object> map = Options.getOptions(args);
032  if (map!= null) {
033    String artifact = generateArtifact(map);
034    emitArtifact(artifact);
035     }
036  
037  }
038 
039  static String generateArtifact(HashMap<String, Object> map) {
040    if(map.containsKey("m")) {
041      PasswordGenerator gen = new SecurePasswordGenerator();
042      for (Map.Entry<String, Object> entry : map.entrySet()) {
043       switch (entry.getKey()) {
044         case "L" :
045          int length = ((Number) entry.getValue()).intValue();
046          gen.setMinLength(length);
047          gen.setMaxLength(length);
048          break;
049         case "a" :
050           gen.setAmbiguousExclude(false);
051           break;
052         case "b" :
053           gen.setUpperCaseInclude (false);
054           break;
055         case "s" :
056           gen.setLowerCaseInclude (false);
057           break;
058         case "n":
059           gen.setNumbersInclude(false);
060           break;
061         case "p":
062           gen.setPunctuationInclude(false);
063           break;
064           default:
065             break;
066       }
067      }
068      //TODO set field for all specified options. 
069      return gen.generate();
070    } else {
071      PassphraseGenerator gen = new PassphraseGenerator();
072      for (Map.Entry<String,Object> entry : map.entrySet()) {
073        switch (entry.getKey()) {
074          case "L":
075          int Length = ((Number) entry.getValue()).intValue();
076          gen.setLength (Length);
077          break;
078          case "d":
079            String delimiter = (String) entry.getValue();
080            gen.setDelimiter(delimiter);
081            break;
082          case "w":
083            String wordListFile = (String) entry.getValue();
084            gen.setWordList(wordListFile);
085            break;
086            default:
087              break;             
088        }
089      }
090      return gen.generate();
091    } 
092  }
093  static void emitArtifact(String artifact) {
094    //TODO make this smarter may check for a verbose option.
095    System.out.println(artifact);
096    
097  }
098}
