075  static HashMap<String, Object> getOptions(String[] args) {
076    org.apache.commons.cli.Options opts = null;
077    ResourceBundle messageBundle = null;   
078    try {
079      messageBundle = ResourceBundle.getBundle(MESSAGES_BUNDLE);
080    } catch (MissingResourceException ex) {
081      System.out.println(FATAL_MESSAGE);
082      return null;
083    }
084    try {
085      opts = buildOptions();
086
087      HashMap<String, Object> map = parseCommandLine(args, opts);
088      if (map != null) {
089        validateCommandLine (map, messageBundle);       
090      }
091//      int length = ((Number) map.get("L")).intValue();
092//      if (cmdLine.hasOption("m")) {
093//        if (length < 5) {
094//          
095//        }
096//      }
097      return map;
098    } catch (MissingArgumentException ex) {
099      Option missing = ex.getOption();
100      String optName = missing.getOpt();
101       displayError(messageBundle, MISSING_ARGUMENT_KEY, opts, optName);
102      return null;
103    } catch (UnrecognizedOptionException ex) {
104      String optName = ex.getOption();
105      displayError(messageBundle, BAD_OPTION_KEY, opts, optName);
106      String message = messageBundle.getString(BAD_OPTION_KEY);
107      message = String.format(message, optName);
108      display (message, usageMessage, opts);
109      return null;
110    } catch (ParseException ex) {
111       displayError(messageBundle, BAD_ARGUMENT_KEY, opts, ex.getLocalizedMessage());
112      return null; 
113    } catch (MissingResourceException ex) {
114      displayError (messageBundle, MISSING_OPTIONS_BUNDLE_KEY, opts, OPTIONS_DESCRIPTION_BUNDLE);
115      return null;
116    } catch (IllegalArgumentException ex){
117      displayError (messageBundle, RESERVED_CHARACTER_ERROR, opts, null);
118      return null;
119    } catch (NegativeArraySizeException ex) {
120      displayError(messageBundle, LENGTH_ERROR, opts, null);
121      return null;
122    } catch (FileNotFoundException ex) {
123      displayError (messageBundle, WORD_LIST_ERROR, opts, ex.getMessage());
124      return null;
125    }
126  }

128  private static void validateCommandLine(HashMap<String, Object>map, ResourceBundle messageBundle) 
129      throws FileNotFoundException, IllegalArgumentException, NegativeArraySizeException {
130    if (map.containsKey("m")) {
131      for (Map.Entry<String, Object> entry : map.entrySet()) {
132        switch (entry.getKey()) {
133          case "L" :
134            int length = ((Number) entry.getValue()).intValue();
135            if (length <= 0) {
136              throw new NegativeArraySizeException();
137            }
138            if (length < MINIMUM_RECOMMENDED_PASSWORD_LENGTH) {
139              System.out.printf(messageBundle.getString(PASSWORD_LENGTH_WARNING), MINIMUM_RECOMMENDED_PASSWORD_LENGTH);
140            } else if (length > MAXIMUM_RECOMMENDED_PASSWORD_LENGTH){
141              System.out.printf(messageBundle.getString(PASSWORD_EXTREME_LENGTH_WARNING),MAXIMUM_RECOMMENDED_PASSWORD_LENGTH);
142            }
143            break;
144          case"a" :
145            System.out.printf(messageBundle.getString(AMBIGUOUS_CHARACTER_WARNING));
146            break;
147          case "d" :
148          case "w" :
149            System.out.printf(messageBundle.getString(PASSWORD_OPTIONS_CONFLICT_WARNING));
150            break;     
151            default :
152              // Do Nothing
153              break;
154        }
155      }
156    } else {
157      for (Map.Entry<String, Object> entry : map.entrySet()) {
158        switch (entry.getKey()) {
159          case "L" :
160            int length = ((Number) entry.getValue()).intValue();
161            if (length <= 0) {
162              throw new NegativeArraySizeException();
163            }
164            if (length < MINIMUM_RECOMMENDED_PASSPHRASE_LENGTH) {
165              System.out.printf(messageBundle.getString(PASSPHRASE_LENGTH_WARNING),MINIMUM_RECOMMENDED_PASSPHRASE_LENGTH );
166            } else if (length > MAXIMUM_RECOMMENDED_PASSPHRASE_LENGTH) {
167              System.out.printf(messageBundle.getString(PASSPHRASE_EXTREME_LENGTH_WARNING),MAXIMUM_RECOMMENDED_PASSPHRASE_LENGTH);
168            }
169            break;
170          case "d" :
171            String delimiter = (String) entry.getValue();
172            if (delimiter.matches(INVALID_DELIMITERS_REGEX)) {
173              throw new IllegalArgumentException();
174            }
175            break;
176          case "w" :
177            String wordListFile = (String) entry.getValue();
178            try {
179             ResourceBundle.getBundle(wordListFile); 
180            } catch (MissingResourceException ex){
181              throw new FileNotFoundException(wordListFile);
182              
183            }
184            break;
185          
186            default :
187              System.out.printf(messageBundle.getString(PASSPHRASE_OPTIONS_CONFLICT_WARNING ));
188            break;
189        }
190      }
191    }
192  }

194  private static HashMap<String, Object> parseCommandLine(String[] args,
195      org.apache.commons.cli.Options opts) throws ParseException {
196    DefaultParser parser = new DefaultParser();
197    HashMap<String, Object> map = new HashMap<>();
198    CommandLine cmdLine = parser.parse(opts, args);
199    if (cmdLine.hasOption("help")){
200      display (null, usageMessage, opts);
201      return null;
202    }
203    
204    for (Option option : cmdLine.getOptions()) {
205      String opt = option.getOpt();
206      map.put(opt, cmdLine.getParsedOptionValue(opt));
207    }
208    return map;
209  }

211  private static org.apache.commons.cli.Options buildOptions() 
212    throws MissingResourceException {
213    ResourceBundle bundle = ResourceBundle.getBundle(OPTIONS_DESCRIPTION_BUNDLE);
214
215    Option helpOption = Option.builder("?").longOpt("help").hasArg(false)
216        .desc(bundle.getString(HELP_OPTION_KEY)).build();
217
218    Option lengthOption =
219        Option.builder("L").argName("length").hasArg()
220                                             .desc(bundle
221                                             .getString(LENGTH_OPTION_KEY))
222                                             .longOpt("length")
223                                             .numberOfArgs(1)
224                                             .type(Number.class)
225                                             .build();
226    Option delimiterOption = Option.builder("d").argName("delimiter")
227                                                .hasArg()
228                                                .desc(bundle.getString(DELIMITER_OPTION_KEY))
229                                                .longOpt("delimiter")
230                                                .numberOfArgs(1)
231                                                .optionalArg(true).type(String.class).build();
232    Option wordListOption = Option.builder("w").argName("path-to-list-file").hasArg()
233        .desc(bundle.getString(WORDLIST_OPTION_KEY)).longOpt("word-list").numberOfArgs(1)
234        .type(String.class).build();
235    Option modeOption = Option.builder("m").longOpt("password-mode").hasArg(false)
236        .desc(bundle.getString(MODE_OPTION_KEY)).build();
237
238    Option excludeUpperOption = Option.builder("b").longOpt("exclude-upper").hasArg(false)
239        .desc(bundle.getString(UPPER_OPTION_KEY)).build();
240    Option excludeLowerOption = Option.builder("s").longOpt("exclude-lower").hasArg(false)
241        .desc(bundle.getString(LOWER_OPTION_KEY)).build();
242    Option excludeDigitsOption = Option.builder("n").longOpt("exclude-digits").hasArg(false)
243        .desc(bundle.getString(DIGITS_OPTION_KEY)).build();
244    Option excludePunctuationOption = Option.builder("p").longOpt("exclude-punctuation")
245        .hasArg(false).desc(bundle.getString(PUNCTUATION_OPTION_KEY)).build();
246    Option includeAmbiguousOption = Option.builder("a").longOpt("include-ambiguous").hasArg(false)
247        .desc(bundle.getString(AMBIGUOUS_OPTION_KEY)).build();
248
249    org.apache.commons.cli.Options opts1 = new org.apache.commons.cli.Options()
250        .addOption(lengthOption).addOption(helpOption).addOption(delimiterOption)
251        .addOption(wordListOption).addOption(modeOption).addOption(excludeUpperOption)
252        .addOption(excludeLowerOption).addOption(excludeDigitsOption)
253        .addOption(excludePunctuationOption).addOption(includeAmbiguousOption);
254    return opts1;
255  }

257  private static void displayError(ResourceBundle messageBundle, String  messageKey,
258      org.apache.commons.cli.Options opts, String optName) {
259    String message = messageBundle.getString(messageKey);
260    message = String.format(message, optName);
261    display (message, usageMessage, opts);
262  }
263  
264  private static void display (String message, String usage, org.apache.commons.cli.Options opts) {
265    if (message != null) {
266      System.out.println(message);
267    }
268    if (opts != null) {
269      new HelpFormatter().printHelp(usage,  opts);
270      
271    }
